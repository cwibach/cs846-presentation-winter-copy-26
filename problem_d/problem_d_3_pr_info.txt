PR Title: Feature: Add JWT-protected /review-audit-log endpoint with role-scoped access control, audit/auth schema expansion, and frontend audit review panel #8

PR Description:
## Summary

  This PR introduces end-to-end audit-log review functionality for the SaaS demo:

  - Adds a new backend endpoint: GET /api/review-audit-log
  - Protects the endpoint using JWT auth (Authorization: Bearer <token>)
  - Enforces role-based visibility for audit logs
  - Expands SQLite schema and seed data to support users, memberships, permissions, and audit logs
  - Adds a frontend “Audit Log Review” panel to authenticate and fetch role-filtered logs

  This implementation keeps the global API prefix (/api) intact and makes the new route available as /api/review-audit-log as requested.

  ## Backend Changes

  ### 1. JWT authentication infrastructure

  Added JWT plumbing under auth/:

  - src/modules/auth/auth.types.ts
      - Defines UserRole, AuthenticatedUser, and JwtPayload
  - src/modules/auth/jwt.service.ts
      - Signs/verifies HS256 tokens
      - Validates header/payload shape and expiry
  - src/modules/auth/jwt-auth.guard.ts
      - Extracts Bearer token
      - Verifies JWT
      - Attaches authenticated user to request

  Wired these into:

  - src/modules/auth/auth.module.ts
      - Registers and exports JwtService + JwtAuthGuard

  ### 2. Auth service/controller updates

  Updated login and current-user flow:

  - src/modules/auth/auth.service.ts
      - POST /auth/login now authenticates against DB (users + organization_memberships)
      - Issues JWT with role + organization context
      - Token TTL set to 8 hours
  - src/modules/auth/auth.controller.ts
      - GET /auth/me now requires JWT (@UseGuards(JwtAuthGuard))

  ### 3. New audit-log review endpoint

  Implemented a dedicated controller:

  - src/modules/usage/review-audit-log.controller.ts
      - GET /review-audit-log
      - JWT-protected
      - Calls UsageService.reviewAuditLog(user)

  Wired in:

  - src/modules/usage/usage.module.ts
      - Imports AuthModule
      - Registers ReviewAuditLogController

  ### 4. Role-scoped audit filtering logic

  Added in src/modules/usage/usage.service.ts:

  - Resolves role permissions from role_audit_permissions
  - Applies one of three scopes:
      - organization: all logs in user org
      - project_assigned: logs only for projects user is assigned to
      - self: logs where user is actor
  - Applies sensitive-data restriction:
      - if can_view_sensitive = 0, filters out is_sensitive = 1
  - Sorts results by occurred_at DESC
  - Throws ForbiddenException if role has no configured audit permission

  ## Database Changes

  ### 1. New schema objects

  Added to problem_d_database/schema.sql:

  - users
  - organization_memberships
  - project_memberships
  - role_audit_permissions
  - audit_logs

  Includes constraints:

  - Role enum check (owner, admin, auditor, member, viewer)
  - Scope enum check (organization, project_assigned, self)
  - Boolean checks for can_view_sensitive / is_sensitive
  - Foreign keys + uniqueness for membership relations

  ### 2. Seed data expansion

  Added to problem_d_database/seed.sql:

  - Sample users across two orgs
  - Organization role assignments
  - Project memberships
  - Role-to-audit permissions mapping
  - Audit log entries (sensitive + non-sensitive, cross-project, cross-org)

  ## Frontend Changes

  ### Added audit review panel in dashboard

  Updated:

  - problem_d_frontend/src/main.ts
  - problem_d_frontend/src/styles.css

  New UI section “Audit Log Review”:

  - Inputs for email/password
  - “Login + Load Logs” action
  - Calls:
      - POST /api/auth/login
      - GET /api/review-audit-log with Bearer token
  - Displays:
      - Authenticated user context (name/role/email)
      - Request status
      - Returned audit rows

  ## API Behavior

  ### New endpoint

  - GET /api/review-audit-log
  - Requires valid JWT
  - Returns only logs the caller’s role is allowed to view

  ### Role visibility model

  - owner, admin:
      - org-wide logs
      - includes sensitive
  - auditor:
      - org-wide logs
      - excludes sensitive
  - member:
      - assigned-project logs only
      - excludes sensitive
  - viewer:
      - own actor logs only
      - excludes sensitive

  ## Security Notes

  - Endpoint is guard-protected and does not trust client-provided role context.
  - Authorization is server-side via DB permission mapping.

+import { CanActivate, ExecutionContext, Injectable, UnauthorizedException } from '@nestjs/common';
+import { AuthenticatedUser } from './auth.types';
+import { JwtService } from './jwt.service';
+
+type RequestWithUser = {
+  headers: Record<string, string | string[] | undefined>;
+  user?: AuthenticatedUser;
+};
+
+@Injectable()
+export class JwtAuthGuard implements CanActivate {
+  constructor(private readonly jwtService: JwtService) {}
+
+  canActivate(context: ExecutionContext): boolean {
+    const request = context.switchToHttp().getRequest<RequestWithUser>();
+    const authHeader = request.headers.authorization;
+    const token = this.extractBearerToken(authHeader);
+
+    if (!token) {
+      throw new UnauthorizedException('Missing Bearer token');
+    }
+
+    const payload = this.jwtService.verify(token);
+    request.user = {
+      id: payload.sub,
+      name: payload.name,
+      email: payload.email,
+      role: payload.role,
+      organizationId: payload.organizationId,
+    };
+
+    return true;
+  }
+
+  private extractBearerToken(authorization: string | string[] | undefined): string | null {
+    const value = Array.isArray(authorization) ? authorization[0] : authorization;
+    if (!value) {
+      return null;
+    }
+
+    const [scheme, token] = value.split(' ');
+    if (scheme !== 'Bearer' || !token) {
+      return null;
+    }
+
+    return token;
+  }

+import { Injectable, UnauthorizedException } from '@nestjs/common';
+import { createHmac, timingSafeEqual } from 'crypto';
+import { JwtPayload } from './auth.types';
+
+type JwtHeader = {
+  alg: 'HS256';
+  typ: 'JWT';
+};
+
+@Injectable()
+export class JwtService {
+  private readonly secret = process.env.JWT_SECRET ?? 'problem-d-demo-secret-change-me';
+
+  sign(payload: JwtPayload): string {
+    const header: JwtHeader = { alg: 'HS256', typ: 'JWT' };
+    const encodedHeader = this.encode(header);
+    const encodedPayload = this.encode(payload);
+    const signature = this.signPart(`${encodedHeader}.${encodedPayload}`);
+    return `${encodedHeader}.${encodedPayload}.${signature}`;
+  }
+
+  verify(token: string): JwtPayload {
+    const [encodedHeader, encodedPayload, signature] = token.split('.');
+    if (!encodedHeader || !encodedPayload || !signature) {
+      throw new UnauthorizedException('Invalid token format');
+    }
+
+    const expectedSignature = this.signPart(`${encodedHeader}.${encodedPayload}`);
+    const signatureMatches =
+      signature.length === expectedSignature.length &&
+      timingSafeEqual(Buffer.from(signature), Buffer.from(expectedSignature));
+
+    if (!signatureMatches) {
+      throw new UnauthorizedException('Invalid token signature');
+    }
+
+    const header = this.decode<JwtHeader>(encodedHeader);
+    if (header.alg !== 'HS256' || header.typ !== 'JWT') {
+      throw new UnauthorizedException('Invalid token header');
+    }
+
+    const payload = this.decode<JwtPayload>(encodedPayload);
+    if (!payload.sub || !payload.role || !payload.organizationId || !payload.exp) {
+      throw new UnauthorizedException('Invalid token payload');
+    }
+
+    return payload;
+  }
+
+  private signPart(part: string): string {
+    return createHmac('sha256', this.secret).update(part).digest('base64url');
+  }
+
+  private encode(value: object): string {
+    return Buffer.from(JSON.stringify(value), 'utf8').toString('base64url');
+  }
+
+  private decode<T>(value: string): T {
+    try {
+      const parsed = JSON.parse(Buffer.from(value, 'base64url').toString('utf8')) as T;
+      return parsed;
+    } catch {
+      throw new UnauthorizedException('Invalid token encoding');
+    }
+  }
+}
